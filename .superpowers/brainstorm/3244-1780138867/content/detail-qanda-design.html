// ComprehensionCheck.tsx — manages revealed answers internally
function ComprehensionCheck({ questions }: { questions: Question[] }) {
  const [revealed, setRevealed] = useState<Set<string>>(new Set());

  const toggleReveal = (questionId: string) => {
    setRevealed(prev => {
      const next = new Set(prev);
      prev.has(questionId) ? next.delete(questionId) : next.add(questionId);
      return next;
    });
  };

  const revealAll = () => setRevealed(new Set(questions.map(q => q.id)));
  const hideAll = () => setRevealed(new Set());

  // ... render list with difficulty badges and show/hide buttons
}

Markdown Rendering

All three text-artifact tabs render as styled markdown via a simple renderer (headings, lists, bold/italic, code). No markdown library dependency — use a lightweight approach:

  • Option: Manual dangerouslySetInnerHTML with a simple regex-based converter (no deps)
  • Alternative: react-markdown (~80KB) if rendering needs get complex
  • Recommendation: Start with a minimal inline renderer; upgrade if needed
