Dagre Layout Configuration

const layoutOptions = {
  rankdir: "LR",       // left-to-right hierarchy
  nodesep: 60,         // horizontal spacing between nodes
  ranksep: 120,        // vertical spacing between ranks
  marginx: 40,
  marginy: 40,
};

// Called after buildGraphViewModel produces nodes + edges:
function applyDagreLayout(
  nodes: GraphNodeRecord[],
  edges: GraphEdgeRecord[]
): { nodes: Node[]; edges: Edge[] } {
  const g = new dagre.graphlib.Graph();
  g.setGraph(layoutOptions);
  g.setDefaultEdgeLabel(() => ({}));
  nodes.forEach(n => g.setNode(n.id, { width: 200, height: 56 }));
  edges.forEach(e => g.setEdge(e.source, e.target));
  dagre.layout(g);
  return {
    nodes: nodes.map(n => ({ ...n, position: g.node(n.id) })),
    edges: edges.map(e => ({ ...e, type: getEdgeType(e.type) })),
  };
}
